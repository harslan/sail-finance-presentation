# Calculate NPV of an investment
cash_flows = [-100000, 30000, 35000, 40000, 45000]
discount_rate = 0.10

npv = sum([cf / (1 + discount_rate)**i for i, cf in enumerate(cash_flows)])
print(f"NPV: ${npv:,.2f}")
